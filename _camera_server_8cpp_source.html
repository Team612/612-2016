58 void CameraServer::SetImage(Image const* image) {
59  unsigned int dataSize = 0;
60  uint8_t* data =
61  (uint8_t*)imaqFlatten(image, IMAQ_FLATTEN_IMAGE, IMAQ_COMPRESSION_JPEG,
62  10 * m_quality, &dataSize);
63 
64  // If we're using a HW camera, then find the start of the data
65  bool hwClient;
66  {
67  // Make a local copy of the hwClient variable so that we can safely use it.
68  std::lock_guard<priority_recursive_mutex> lock(m_imageMutex);
69  hwClient = m_hwClient;
70  }
71  unsigned int start = 0;
72  if (hwClient) {
73  while (start < dataSize - 1) {
74  if (data[start] == 0xFF && data[start + 1] == 0xD8)
75  break;
76  else
77  start++;
78  }
79  }
80  dataSize -= start;
81 
82  wpi_assert(dataSize > 2);
83  SetImageData(data, dataSize, start, true);
84 }

86 void CameraServer::AutoCapture() {
87  Image* frame = imaqCreateImage(IMAQ_IMAGE_RGB, 0);
88 
89  while (true) {
90  bool hwClient;
91  uint8_t* data = nullptr;
92  {
93  std::lock_guard<priority_recursive_mutex> lock(m_imageMutex);
94  hwClient = m_hwClient;
95  if (hwClient) {
96  data = m_dataPool.back();
97  m_dataPool.pop_back();
98  }
99  }
100 
101  if (hwClient) {
102  unsigned int size = m_camera->GetImageData(data, kMaxImageSize);
103  SetImageData(data, size);
104  } else {
105  m_camera->GetImage(frame);
106  SetImage(frame);
107  }
108  }
109 }

156 void CameraServer::Serve() {
157  int sock = socket(AF_INET, SOCK_STREAM, 0);
158 
159  if (sock == -1) {
160  wpi_setErrnoError();
161  return;
162  }
163 
164  int reuseAddr = 1;
165  if (setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, &reuseAddr,
166  sizeof(reuseAddr)) == -1)
167  wpi_setErrnoError();
168 
169  sockaddr_in address, clientAddress;
170 
171  memset(&address, 0, sizeof(address));
172  address.sin_family = AF_INET;
173  address.sin_addr.s_addr = htonl(INADDR_ANY);
174  address.sin_port = htons(kPort);
175 
176  if (bind(sock, (struct sockaddr*)&address, sizeof(address)) == -1)
177  wpi_setErrnoError();
178 
179  if (listen(sock, 10) == -1) wpi_setErrnoError();
180 
181  while (true) {
182  socklen_t clientAddressLen = sizeof(clientAddress);
183 
184  int conn =
185  accept(sock, (struct sockaddr*)&clientAddress, &clientAddressLen);
186  if (conn == -1) {
187  wpi_setErrnoError();
188  continue;
189  }
190 
191  Request req;
192  char reqBuf[sizeof(req)];
193  size_t reqPos = 0;
194 
195  while (reqPos < sizeof(req)) {
196  ssize_t sizeRead = read(conn, &reqBuf[reqPos], sizeof(req) - reqPos);
197  if (sizeRead < 0) break;
198  reqPos += sizeRead;
199  }
200 
201  if (reqPos < sizeof(req)) {
202  wpi_setErrnoError();
203  close(conn);
204  continue;
205  }
206 
207  memcpy(&req, reqBuf, sizeof(req));
208  req.fps = ntohl(req.fps);
209  req.compression = ntohl(req.compression);
210  req.size = ntohl(req.size);
211 
212  // TODO: Support the SW Compression. The rest of the code below will work as
213  // though this
214  // check isn't here
215  if (req.compression != kHardwareCompression) {
216  wpi_setWPIErrorWithContext(IncompatibleState,
217  "Choose \"USB Camera HW\" on the dashboard");
218  close(conn);
219  continue;
220  }
221 
222  {
223  // Wait for the camera to be setw
224  std::unique_lock<priority_recursive_mutex> lock(m_imageMutex);
225  if (!m_camera) {
226  std::cout << "Camera not yet ready, awaiting first image" << std::endl;
227  m_newImageVariable.wait(lock);
228  }
229  m_hwClient = req.compression == kHardwareCompression;
230  if (!m_hwClient)
231  SetQuality(100 - req.compression);
232  else if (m_camera)
233  m_camera->SetFPS(req.fps);
234  SetSize(req.size);
235  }
236 
237  auto period = std::chrono::microseconds(1000000) / req.fps;
238  while (true) {
239  auto startTime = std::chrono::steady_clock::now();
240  std::tuple<uint8_t*, unsigned int, unsigned int, bool> imageData;
241  {
242  std::unique_lock<priority_recursive_mutex> lock(m_imageMutex);
243  m_newImageVariable.wait(lock);
244  imageData = m_imageData;
245  m_imageData = std::make_tuple<uint8_t*>(nullptr, 0, 0, false);
246  }
247 
248  unsigned int size = std::get<1>(imageData);
249  unsigned int netSize = htonl(size);
250  unsigned int start = std::get<2>(imageData);
251  uint8_t* data = std::get<0>(imageData);
252 
253  if (data == nullptr) continue;
254 
255  if (write(conn, kMagicNumber, sizeof(kMagicNumber)) == -1) {
256  wpi_setErrnoErrorWithContext(
257  "[CameraServer] Error sending magic number");
258  FreeImageData(imageData);
259  break;
260  }
261  if (write(conn, &netSize, sizeof(netSize)) == -1) {
262  wpi_setErrnoErrorWithContext("[CameraServer] Error sending image size");
263  FreeImageData(imageData);
264  break;
265  }
266  if (write(conn, &data[start], sizeof(uint8_t) * size) == -1) {
267  wpi_setErrnoErrorWithContext("[CameraServer] Error sending image data");
268  FreeImageData(imageData);
269  break;
270  }
271  FreeImageData(imageData);
272  std::this_thread::sleep_until(startTime + period);
273  }
274  close(conn);
275  }
276  close(sock);
277 }
