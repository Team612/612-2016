27 /**
28  * Initialize the gyro.
29  * Calibrate the gyro by running for a number of samples and computing the
30  * center value.
31  * Then use the center value as the Accumulator center value for subsequent
32  * measurements.
33  * It's important to make sure that the robot is not moving while the centering
34  * calculations are in progress, this is typically done when the robot is first
35  * turned on while it's sitting at rest before the competition starts.
36  */
37 void ADXRS450_Gyro::Calibrate() {
38  Wait(0.1);
39 
40  m_spi.SetAccumulatorCenter(0);
41  m_spi.ResetAccumulator();
42 
43  Wait(kCalibrationSampleTime);
44 
45  m_spi.SetAccumulatorCenter((int32_t)m_spi.GetAccumulatorAverage());
46  m_spi.ResetAccumulator();
47 }

59 ADXRS450_Gyro::ADXRS450_Gyro(SPI::Port port) : m_spi(port) {
60  m_spi.SetClockRate(3000000);
61  m_spi.SetMSBFirst();
62  m_spi.SetSampleDataOnRising();
63  m_spi.SetClockActiveHigh();
64  m_spi.SetChipSelectActiveLow();
65 
66  // Validate the part ID
67  if ((ReadRegister(kPIDRegister) & 0xff00) != 0x5200) {
68  DriverStation::ReportError("could not find ADXRS450 gyro");
69  return;
70  }
71 
72  m_spi.InitAccumulator(kSamplePeriod, 0x20000000u, 4, 0x0c000000u, 0x04000000u,
73  10u, 16u, true, true);
74 
75  Calibrate();
76 
77  HALReport(HALUsageReporting::kResourceType_ADXRS450, port);
78  LiveWindow::GetInstance()->AddSensor("ADXRS450_Gyro", port, this);
79 }

81 static bool CalcParity(uint32_t v) {
82  bool parity = false;
83  while (v != 0) {
84  parity = !parity;
85  v = v & (v - 1);
86  }
87  return parity;
88 }

90 static inline uint32_t BytesToIntBE(uint8_t* buf) {
91  uint32_t result = ((uint32_t)buf[0]) << 24;
92  result |= ((uint32_t)buf[1]) << 16;
93  result |= ((uint32_t)buf[2]) << 8;
94  result |= (uint32_t)buf[3];
95  return result;
96 }

98 uint16_t ADXRS450_Gyro::ReadRegister(uint8_t reg) {
99  uint32_t cmd = 0x80000000 | (((uint32_t)reg) << 17);
100  if (!CalcParity(cmd)) cmd |= 1u;
101 
102  // big endian
103  uint8_t buf[4] = {(uint8_t)((cmd >> 24) & 0xff),
104  (uint8_t)((cmd >> 16) & 0xff),
105  (uint8_t)((cmd >> 8) & 0xff),
106  (uint8_t)(cmd & 0xff)};
107 
108  m_spi.Write(buf, 4);
109  m_spi.Read(false, buf, 4);
110  if ((buf[0] & 0xe0) == 0) return 0; // error, return 0
111  return (uint16_t)((BytesToIntBE(buf) >> 5) & 0xffff);
112 }

124 /**
125  * Return the actual angle in degrees that the robot is currently facing.
126  *
127  * The angle is based on the current accumulator value corrected by the
128  * oversampling rate, the
129  * gyro type and the A/D calibration values.
130  * The angle is continuous, that is it will continue from 360->361 degrees. This
131  * allows algorithms that wouldn't
132  * want to see a discontinuity in the gyro output as it sweeps from 360 to 0 on
133  * the second time around.
134  *
135  * @return the current heading of the robot in degrees. This heading is based on
136  * integration
137  * of the returned rate from the gyro.
138  */
139 float ADXRS450_Gyro::GetAngle() const {
140  return (float)(m_spi.GetAccumulatorValue() * kDegreePerSecondPerLSB *
141  kSamplePeriod);
142 }
