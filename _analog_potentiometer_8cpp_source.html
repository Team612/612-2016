54 /**
55  * Get the current reading of the potentiometer.
56  *
57  * @return The current position of the potentiometer (in the units used for
58  * fullRaneg and offset).
59  */
60 double AnalogPotentiometer::Get() const {
61  return (m_analog_input->GetVoltage() / ControllerPower::GetVoltage5V()) *
62  m_fullRange +
63  m_offset;
64 }
