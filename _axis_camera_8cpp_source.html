107 /**
108  * Copy an image into an existing buffer.
109  * This copies an image into an existing buffer rather than creating a new image
110  * in memory. That way a new image is only allocated when the image being copied
111  * is
112  * larger than the destination.
113  * This method is called by the PCVideoServer class.
114  * @param imageData The destination image.
115  * @param numBytes The size of the destination image.
116  * @return 0 if failed (no source image or no memory), 1 if success.
117  */
118 int AxisCamera::CopyJPEG(char **destImage, unsigned int &destImageSize,
119  unsigned int &destImageBufferSize) {
120  std::lock_guard<priority_mutex> lock(m_imageDataMutex);
121  if (destImage == nullptr) {
122  wpi_setWPIErrorWithContext(NullParameter, "destImage must not be nullptr");
123  return 0;
124  }
125 
126  if (m_imageData.size() == 0) return 0; // if no source image
127 
128  if (destImageBufferSize <
129  m_imageData.size()) // if current destination buffer too small
130  {
131  if (*destImage != nullptr) delete[] * destImage;
132  destImageBufferSize = m_imageData.size() + kImageBufferAllocationIncrement;
133  *destImage = new char[destImageBufferSize];
134  if (*destImage == nullptr) return 0;
135  }
136  // copy this image into destination buffer
137  if (*destImage == nullptr) {
138  wpi_setWPIErrorWithContext(NullParameter, "*destImage must not be nullptr");
139  }
140 
141  std::copy(m_imageData.begin(), m_imageData.end(), *destImage);
142  destImageSize = m_imageData.size();
143  ;
144  return 1;
145 }

375 /**
376  * Method called in the capture thread to receive images from the camera
377  */
378 void AxisCamera::Capture() {
379  int consecutiveErrors = 0;
380 
381  // Loop on trying to setup the camera connection. This happens in a background
382  // thread so it shouldn't effect the operation of user programs.
383  while (!m_done) {
384  std::string requestString =
385  "GET /mjpg/video.mjpg HTTP/1.1\n"
386  "User-Agent: HTTPStreamClient\n"
387  "Connection: Keep-Alive\n"
388  "Cache-Control: no-cache\n"
389  "Authorization: Basic RlJDOkZSQw==\n\n";
390  m_captureMutex.lock();
391  m_cameraSocket = CreateCameraSocket(requestString, consecutiveErrors > 5);
392  if (m_cameraSocket != -1) {
393  ReadImagesFromCamera();
394  consecutiveErrors = 0;
395  } else {
396  consecutiveErrors++;
397  }
398  m_captureMutex.unlock();
399  Wait(0.5);
400  }
401 }

403 /**
404  * This function actually reads the images from the camera.
405  */
406 void AxisCamera::ReadImagesFromCamera() {
407  char *imgBuffer = nullptr;
408  int imgBufferLength = 0;
409 
410  // TODO: these recv calls must be non-blocking. Otherwise if the camera
411  // fails during a read, the code hangs and never retries when the camera comes
412  // back up.
413 
414  int counter = 2;
415  while (!m_done) {
416  char initialReadBuffer[kMaxPacketSize] = "";
417  char intermediateBuffer[1];
418  char *trailingPtr = initialReadBuffer;
419  int trailingCounter = 0;
420  while (counter) {
421  // TODO: fix me... this cannot be the most efficient way to approach this,
422  // reading one byte at a time.
423  if (recv(m_cameraSocket, intermediateBuffer, 1, 0) == -1) {
424  wpi_setErrnoErrorWithContext("Failed to read image header");
425  close(m_cameraSocket);
426  return;
427  }
428  strncat(initialReadBuffer, intermediateBuffer, 1);
429  // trailingCounter ensures that we start looking for the 4 byte string
430  // after
431  // there is at least 4 bytes total. Kind of obscure.
432  // look for 2 blank lines (\r\n)
433  if (nullptr != strstr(trailingPtr, "\r\n\r\n")) {
434  --counter;
435  }
436  if (++trailingCounter >= 4) {
437  trailingPtr++;
438  }
439  }
440  counter = 1;
441  char *contentLength = strstr(initialReadBuffer, "Content-Length: ");
442  if (contentLength == nullptr) {
443  wpi_setWPIErrorWithContext(IncompatibleMode,
444  "No content-length token found in packet");
445  close(m_cameraSocket);
446  if (imgBuffer) delete[] imgBuffer;
447  return;
448  }
449  contentLength = contentLength + 16; // skip past "content length"
450  int readLength = atol(contentLength); // get the image byte count
451 
452  // Make sure buffer is large enough
453  if (imgBufferLength < readLength) {
454  if (imgBuffer) delete[] imgBuffer;
455  imgBufferLength = readLength + kImageBufferAllocationIncrement;
456  imgBuffer = new char[imgBufferLength];
457  if (imgBuffer == nullptr) {
458  imgBufferLength = 0;
459  continue;
460  }
461  }
462 
463  // Read the image data for "Content-Length" bytes
464  int bytesRead = 0;
465  int remaining = readLength;
466  while (bytesRead < readLength) {
467  int bytesThisRecv =
468  recv(m_cameraSocket, &imgBuffer[bytesRead], remaining, 0);
469  bytesRead += bytesThisRecv;
470  remaining -= bytesThisRecv;
471  }
472 
473  // Update image
474  {
475  std::lock_guard<priority_mutex> lock(m_imageDataMutex);
476 
477  m_imageData.assign(imgBuffer, imgBuffer + imgBufferLength);
478  m_freshImage = true;
479  }
480 
481  if (WriteParameters()) {
482  break;
483  }
484  }
485 
486  close(m_cameraSocket);
487 }

489 /**
490  * Send a request to the camera to set all of the parameters. This is called
491  * in the capture thread between each frame. This strategy avoids making lots
492  * of redundant HTTP requests, accounts for failed initial requests, and
493  * avoids blocking calls in the main thread unless necessary.
494  *
495  * This method does nothing if no parameters have been modified since it last
496  * completely successfully.
497  *
498  * @return <code>true</code> if the stream should be restarted due to a
499  * parameter changing.
500  */
501 bool AxisCamera::WriteParameters() {
502  if (m_parametersDirty) {
503  std::stringstream request;
504  request << "GET /axis-cgi/admin/param.cgi?action=update";
505 
506  m_parametersMutex.lock();
507  request << "&ImageSource.I0.Sensor.Brightness=" << m_brightness;
508  request << "&ImageSource.I0.Sensor.WhiteBalance="
509  << kWhiteBalanceStrings[m_whiteBalance];
510  request << "&ImageSource.I0.Sensor.ColorLevel=" << m_colorLevel;
511  request << "&ImageSource.I0.Sensor.Exposure="
512  << kExposureControlStrings[m_exposureControl];
513  request << "&ImageSource.I0.Sensor.ExposurePriority=" << m_exposurePriority;
514  request << "&Image.I0.Stream.FPS=" << m_maxFPS;
515  request << "&Image.I0.Appearance.Resolution="
516  << kResolutionStrings[m_resolution];
517  request << "&Image.I0.Appearance.Compression=" << m_compression;
518  request << "&Image.I0.Appearance.Rotation=" << kRotationStrings[m_rotation];
519  m_parametersMutex.unlock();
520 
521  request << " HTTP/1.1" << std::endl;
522  request << "User-Agent: HTTPStreamClient" << std::endl;
523  request << "Connection: Keep-Alive" << std::endl;
524  request << "Cache-Control: no-cache" << std::endl;
525  request << "Authorization: Basic RlJDOkZSQw==" << std::endl;
526  request << std::endl;
527 
528  int socket = CreateCameraSocket(request.str(), false);
529  if (socket == -1) {
530  wpi_setErrnoErrorWithContext("Error setting camera parameters");
531  } else {
532  close(socket);
533  m_parametersDirty = false;
534 
535  if (m_streamDirty) {
536  m_streamDirty = false;
537  return true;
538  }
539  }
540  }
541 
542  return false;
543 }

545 /**
546  * Create a socket connected to camera
547  * Used to create a connection to the camera for both capturing images and
548  * setting parameters.
549  * @param requestString The initial request string to send upon successful
550  * connection.
551  * @param setError If true, rais an error if there's a problem creating the
552  * connection.
553  * This is only enabled after several unsucessful connections, so a single one
554  * doesn't
555  * cause an error message to be printed if it immediately recovers.
556  * @return -1 if failed, socket handle if successful.
557  */
558 int AxisCamera::CreateCameraSocket(std::string const &requestString,
559  bool setError) {
560  struct addrinfo *address = nullptr;
561  int camSocket;
562 
563  /* create socket */
564  if ((camSocket = socket(AF_INET, SOCK_STREAM, 0)) == -1) {
565  if (setError)
566  wpi_setErrnoErrorWithContext("Failed to create the camera socket");
567  return -1;
568  }
569 
570  if (getaddrinfo(m_cameraHost.c_str(), "80", nullptr, &address) == -1) {
571  if (setError) {
572  wpi_setErrnoErrorWithContext("Failed to create the camera socket");
573  close(camSocket);
574  }
575  return -1;
576  }
577 
578  /* connect to server */
579  if (connect(camSocket, address->ai_addr, address->ai_addrlen) == -1) {
580  if (setError)
581  wpi_setErrnoErrorWithContext("Failed to connect to the camera");
582  freeaddrinfo(address);
583  close(camSocket);
584  return -1;
585  }
586 
587  freeaddrinfo(address);
588 
589  int sent = send(camSocket, requestString.c_str(), requestString.size(), 0);
590  if (sent == -1) {
591  if (setError)
592  wpi_setErrnoErrorWithContext("Failed to send a request to the camera");
593  close(camSocket);
594  return -1;
595  }
596 
597  return camSocket;
598 }
